    Performance Tuning¶

    This guide covers optimization strategies, memory management, and performance best practices for using kmeanssa-ng efficiently with large datasets and complex graphs.

    Understanding Performance Bottlenecks¶

    Profiling Your Clustering Tasks¶

    from kmeanssa_ng import generate_sbm, SimulatedAnnealing, QGSimulatedAnnealing

def profile_memory_usage():
    """Get current memory usage of the process."""
    process = psutil.Process(os.getpid())
    return process.memory_info().rss / 1024 / 1024  # MB

def benchmark_clustering(graph, n_points, k, description):
    """Benchmark clustering performance with detailed metrics."""

    print(f"=== {description} ===")

    # Memory before
    mem_before = profile_memory_usage()

    # Sample points
    start_time = time.time()
    points = graph.sample_points(n_points)
    sampling_time = time.time() - start_time
    mem_after_sampling = profile_memory_usage()

    # Run clustering
    start_time = time.time()
    sa = SimulatedAnnealing(points, k=k)
    centers = sa.run(initialization="kpp", robust_prop=0.1)
    clustering_time = time.time() - start_time
    mem_after_clustering = profile_memory_usage()

    # Calculate energy
    start_time = time.time()
    energy = sa.calculate_energy(centers, points)
    energy_time = time.time() - start_time

    # Report results
    print(f"Graph: {len(graph.nodes())} nodes, {len(graph.edges())} edges")
    print(f"Points: {n_points}, Clusters: {k}")
    print(f"Timings:")
    print(f"  Sampling: {sampling_time:.3f}s")
    print(f"  Clustering: {clustering_time:.3f}s")
    print(f"  Energy calc: {energy_time:.3f}s")
    print(f"  Total: {sampling_time + clustering_time + energy_time:.3f}s")
    print(f"Memory usage:")
    print(f"  Initial: {mem_before:.1f} MB")
    print(f"  After sampling: {mem_after_sampling:.1f} MB (+{mem_after_sampling - mem_before:.1f})")
    print(f"  After clustering: {mem_after_clustering:.1f} MB (+{mem_after_clustering - mem_after_sampling:.1f})")
    print(f"Final energy: {energy:.3f}")

        'energy_time': energy_time,

    def optimize_graph_structure(graph, max_nodes=1000):
    """Optimize graph structure for better clustering performance."""

    original_nodes = len(graph.nodes())
    original_edges = len(graph.edges())

    print(f"Original graph: {original_nodes} nodes, {original_edges} edges")

    # Strategy 1: Work with largest connected component
    if not nx.is_connected(graph):
        largest_cc = max(nx.connected_components(graph), key=len)
        graph = graph.subgraph(largest_cc).copy()
        print(f"Extracted largest component: {len(graph.nodes())} nodes")

    # Strategy 2: Sample nodes if graph is too large
    if len(graph.nodes()) > max_nodes:
        # Sample nodes while preserving connectivity
        sampled_nodes = list(graph.nodes())[:max_nodes]
        graph = graph.subgraph(sampled_nodes).copy()

        # Ensure connectivity
        if not nx.is_connected(graph):
            largest_cc = max(nx.connected_components(graph), key=len)
            graph = graph.subgraph(largest_cc).copy()

        print(f"Sampled to {len(graph.nodes())} nodes")

    # Strategy 3: Precompute distances
    graph.precomputing()
    print("Distance precomputation completed")

    # Strategy 4: Validate edge weights for sampling efficiency
    edge_lengths = [d['length'] for u, v, d in graph.edges(data=True)]
    if edge_lengths:
        avg_length = np.mean(edge_lengths)
        std_length = np.std(edge_lengths)
        print(f"Edge lengths: μ={avg_length:.3f}, σ={std_length:.3f}")

        # Normalize extreme lengths if needed
        if std_length > 10 * avg_length:
            print("Warning: High variance in edge lengths may affect performance")

    final_nodes = len(graph.nodes())
    final_edges = len(graph.edges())
    reduction_factor = original_nodes / final_nodes if final_nodes > 0 else 1

    print(f"Final graph: {final_nodes} nodes, {final_edges} edges")
    print(f"Reduction factor: {reduction_factor:.2f}x")
    print()

    return graph

    Parameter Optimization for Speed¶

    def get_fast_parameters(n_points, k, graph_density=0.1):
    """Get parameter settings optimized for speed."""

    base_params = {
        'lambda_param': 1.0,
        'beta': 1.0, 
        'step_size': 0.1
    }

    # Adjust for dataset size
    if n_points > 500:
        base_params['lambda_param'] = 2.0  # Higher exploration for large datasets
        base_params['step_size'] = 0.15    # Larger steps for faster convergence
    elif n_points < 100:
        base_params['beta'] = 2.0          # Faster convergence for small datasets
        base_params['step_size'] = 0.05    # Smaller steps for precision

    # Adjust for number of clusters
    if k > 5:
        base_params['lambda_param'] *= 1.5  # More exploration for many clusters

    # Adjust for graph density
    if graph_density > 0.2:  # Dense graph
        base_params['step_size'] *= 0.8    # Smaller steps in dense graphs
    elif graph_density < 0.05:  # Sparse graph
        base_params['step_size'] *= 1.3    # Larger steps in sparse graphs

    return base_params

    Memory Optimization¶

    def memory_efficient_clustering(graph, n_points, k, batch_size=100):
    """Perform clustering with memory optimization techniques."""

    print(f"=== Memory-Efficient Clustering ===")
    print(f"Processing {n_points} points in batches of {batch_size}")

    mem_start = profile_memory_usage()

    # Strategy 1: Batch processing of points
    all_points = []
    for batch_start in range(0, n_points, batch_size):
        batch_end = min(batch_start + batch_size, n_points)
        batch_points = graph.sample_points(batch_end - batch_start)
        all_points.extend(batch_points)

        if batch_start % (batch_size * 3) == 0:  # Report every 3 batches
            mem_current = profile_memory_usage()
            print(f"Batch {batch_start//batch_size + 1}: {mem_current:.1f} MB (+{mem_current - mem_start:.1f})")

    mem_after_sampling = profile_memory_usage()
    print(f"Sampling complete: {mem_after_sampling:.1f} MB")

    # Strategy 2: Use smaller robustification for memory efficiency
    sa = SimulatedAnnealing(all_points, k=k, step_size=0.15)  # Larger steps = faster = less memory over time
    centers = sa.run(robust_prop=0.05, initialization="kpp")  # Smaller robustification

    mem_after_clustering = profile_memory_usage()
    print(f"Clustering complete: {mem_after_clustering:.1f} MB")

    # Strategy 3: Incremental energy calculation
    total_energy = 0.0
    energy_batch_size = batch_size // 2

    for i in range(0, len(all_points), energy_batch_size):
        batch_points = all_points[i:i + energy_batch_size]
        batch_energy = sa.calculate_energy(centers, batch_points)
        total_energy += batch_energy * len(batch_points)

    average_energy = total_energy / len(all_points)

    mem_final = profile_memory_usage()
    print(f"Final memory: {mem_final:.1f} MB")
    print(f"Peak memory usage: {max(mem_after_sampling, mem_after_clustering, mem_final):.1f} MB")
    print(f"Average energy: {average_energy:.3f}")

    return centers, average_energy

    def parallel_initialization_comparison(points, k, n_trials=5):
    """Compare different initialization strategies for consistency and speed."""

    print(f"=== Initialization Strategy Comparison ===")
    print(f"Testing with {len(points)} points, k={k}, {n_trials} trials each")

    strategies = ['random', 'kpp']
    results = {}

    for strategy in strategies:
        energies = []
        times = []

        print(f"\nTesting {strategy} initialization:")

        for trial in range(n_trials):
            start_time = time.time()
            sa = SimulatedAnnealing(points, k=k)
            centers = sa.run(initialization=strategy, robust_prop=0.1)
            end_time = time.time()

            energy = sa.calculate_energy(centers, points)
            runtime = end_time - start_time

            energies.append(energy)
            times.append(runtime)

            if trial < 3:  # Show first few trials
                print(f"  Trial {trial + 1}: {runtime:.3f}s, energy={energy:.3f}")

        # Statistics
        mean_energy = np.mean(energies)
        std_energy = np.std(energies)
        mean_time = np.mean(times)
        std_time = np.std(times)

        results[strategy] = {
            'mean_energy': mean_energy,
            'std_energy': std_energy,
            'mean_time': mean_time,
            'std_time': std_time,
            'consistency': std_energy / mean_energy  # Lower is more consistent
        }

        print(f"  Average: {mean_time:.3f}±{std_time:.3f}s, energy={mean_energy:.3f}±{std_energy:.3f}")
        print(f"  Consistency score: {results[strategy]['consistency']:.4f}")

    # Recommendation
    if results['kpp']['consistency'] < results['random']['consistency']:
        print(f"\nRecommendation: Use k-means++ for better consistency")
    else:
        print(f"\nRecommendation: Random initialization is sufficient")

    if results['kpp']['mean_time'] > results['random']['mean_time'] * 1.5:
        print(f"Note: k-means++ is {results['kpp']['mean_time'] / results['random']['mean_time']:.1f}x slower")

    return results

    def compare_algorithm_versions(points, k):
    """Compare algorithm versions for different scenarios."""

    print(f"=== Algorithm Version Comparison ===")
    print(f"Dataset: {len(points)} points, k={k}")

    versions = ['v1', 'v2']
    version_results = {}

    for version in versions:
        print(f"\nTesting algorithm {version}:")

        # Multiple runs for stability
        energies = []
        times = []

        for run in range(3):
            start_time = time.time()
            sa = SimulatedAnnealing(points, k=k)
            centers = sa.run(
                algorithm_version=version,
                robust_prop=0.1,
                initialization="kpp"
            )
            end_time = time.time()

            energy = sa.calculate_energy(centers, points)
            runtime = end_time - start_time

            energies.append(energy)
            times.append(runtime)

            print(f"  Run {run + 1}: {runtime:.3f}s, energy={energy:.3f}")

        mean_energy = np.mean(energies)
        mean_time = np.mean(times)

        version_results[version] = {
            'mean_energy': mean_energy,
            'mean_time': mean_time,
            'energies': energies,
            'times': times
        }

        print(f"  Average: {mean_time:.3f}s, energy={mean_energy:.3f}")

    # Analysis and recommendation
    v1_result = version_results['v1']
    v2_result = version_results['v2']

    speed_ratio = v1_result['mean_time'] / v2_result['mean_time']
    energy_ratio = v1_result['mean_energy'] / v2_result['mean_energy']

    print(f"\n=== Comparison Analysis ===")
    print(f"Speed: v1 vs v2 = {speed_ratio:.2f} (lower is faster)")
    print(f"Energy: v1 vs v2 = {energy_ratio:.3f} (lower is better)")

    if speed_ratio < 0.9:
        print("Recommendation: Use v1 (interleaved) for better speed")
    elif speed_ratio > 1.1:
        print("Recommendation: Use v2 (sequential) for better speed")
    else:
        print("Recommendation: Both versions have similar performance")

    if abs(energy_ratio - 1.0) > 0.05:
        better_quality = "v1" if energy_ratio < 1.0 else "v2"
        print(f"Note: {better_quality} provides better clustering quality")

    return version_results

    Monitoring and Debugging¶

    def create_performance_monitor():
    """Create a simple performance monitoring utility."""

    class PerformanceMonitor:
        def __init__(self):
            self.start_time = None
            self.start_memory = None
            self.checkpoints = []

        def start(self):
            self.start_time = time.time()
            self.start_memory = profile_memory_usage()
            self.checkpoints = []
            print(f"Performance monitoring started (Memory: {self.start_memory:.1f} MB)")

        def checkpoint(self, name):
            current_time = time.time()
            current_memory = profile_memory_usage()

            elapsed = current_time - self.start_time
            memory_delta = current_memory - self.start_memory

            self.checkpoints.append({
                'name': name,
                'elapsed': elapsed,
                'memory': current_memory,
                'memory_delta': memory_delta
            })

            print(f"Checkpoint '{name}': {elapsed:.3f}s, {current_memory:.1f} MB (+{memory_delta:+.1f})")

        def finish(self):
            if self.checkpoints:
                total_time = self.checkpoints[-1]['elapsed']
                peak_memory = max(cp['memory'] for cp in self.checkpoints)
                print(f"Monitoring complete: {total_time:.3f}s total, {peak_memory:.1f} MB peak")
            return self.checkpoints

    return PerformanceMonitor()
