    Custom Metric Spaces¶

    While quantum graphs are powerful, you can extend kmeanssa-ng to work with any metric space by implementing the abstract base classes. This guide shows you how to create custom metric spaces for specialized clustering applications.

    Understanding the Architecture¶

    The library uses a three-layer abstraction pattern:

    1. Point - Immutable elements in your metric space
    2. Center - Movable points that serve as cluster centers
    3. Space - The metric space containing points and distance computation
    from kmeanssa_ng.core.abstract import Point, Center, Space
from kmeanssa_ng import SimulatedAnnealing
import numpy as np
import math

print("Abstract base classes imported successfully!")

    Abstract base classes imported successfully!

    Creating a Simple 1D Metric Space¶

    Let’s start with a basic example - clustering on a line (1D space):

    # 1D Point implementation
class LinePoint(Point):
    def __init__(self, space, x: float):
        self._space = space
        self.x = x

    @property 
    def space(self):
        return self._space

    def __repr__(self):
        return f"LinePoint(x={self.x:.2f})"

    def __eq__(self, other):
        return isinstance(other, LinePoint) and abs(self.x - other.x) < 1e-10

    def __hash__(self):
        return hash(round(self.x, 10))

# 1D Center implementation  
class LineCenter(LinePoint, Center):
    def __init__(self, space, x: float):
        super().__init__(space, x)

    def brownian_motion(self, time_to_travel: float):
        """Random walk along the line."""
        step = np.random.normal(0, math.sqrt(time_to_travel))
        self.x += step

        # Keep within space bounds
        self.x = max(self._space.min_x, min(self._space.max_x, self.x))

    def drift(self, target_point: Point, prop_to_travel: float):
        """Move toward target point along the line."""
        if not isinstance(target_point, LinePoint):
            raise ValueError("Target must be a LinePoint")

        direction = target_point.x - self.x
        self.x += direction * prop_to_travel

    def __repr__(self):
        return f"LineCenter(x={self.x:.2f})"

# 1D Space implementation
class LineSpace(Space):
    def __init__(self, min_x: float = 0.0, max_x: float = 10.0):
        self.min_x = min_x
        self.max_x = max_x

    def distance(self, p1: Point, p2: Point) -> float:
        """Euclidean distance on the line."""
        if not isinstance(p1, LinePoint) or not isinstance(p2, LinePoint):
            raise ValueError("Points must be LinePoint instances")
        return abs(p1.x - p2.x)

    def sample_points(self, n: int) -> list[Point]:
        """Sample random points uniformly on the line."""
        points = []
        for _ in range(n):
            x = np.random.uniform(self.min_x, self.max_x)
            points.append(LinePoint(self, x))
        return points

    def sample_centers(self, k: int) -> list[Center]:
        """Sample random centers uniformly on the line."""
        centers = []
        for _ in range(k):
            x = np.random.uniform(self.min_x, self.max_x)
            centers.append(LineCenter(self, x))
        return centers

    def sample_kpp_centers(self, k: int) -> list[Center]:
        """k-means++ initialization for 1D space."""
        return self.sample_centers(k)  # Simple fallback for 1D

    def compute_clusters(self, centers: list[Center]):
        """Not needed for basic usage."""
        pass

    def calculate_energy_graph(self, centers: list[Center]) -> float:
        """Calculate clustering energy (not used in this example)."""
        return 0.0

    def batch_distances_from_centers(self, centers: list[Center], point: Point) -> list[float]:
        """Compute distances from a point to all centers."""
        return [self.distance(point, center) for center in centers]

    Advanced Example: 2D Euclidean Space¶

    Let’s create a more sophisticated 2D space with custom distance metrics:

    # 2D Point implementation
class Point2D(Point):
    def __init__(self, space, x: float, y: float):
        self._space = space
        self.x = x
        self.y = y

    @property
    def space(self):
        return self._space

    def __repr__(self):
        return f"Point2D({self.x:.2f}, {self.y:.2f})"

    def __eq__(self, other):
        return (isinstance(other, Point2D) and 
                abs(self.x - other.x) < 1e-10 and 
                abs(self.y - other.y) < 1e-10)

    def __hash__(self):
        return hash((round(self.x, 10), round(self.y, 10)))

# 2D Center implementation
class Center2D(Point2D, Center):
    def brownian_motion(self, time_to_travel: float):
        """2D random walk."""
        step_x = np.random.normal(0, math.sqrt(time_to_travel))
        step_y = np.random.normal(0, math.sqrt(time_to_travel))

        self.x += step_x
        self.y += step_y

        # Keep within bounds
        self.x = max(self._space.min_x, min(self._space.max_x, self.x))
        self.y = max(self._space.min_y, min(self._space.max_y, self.y))

    def drift(self, target_point: Point, prop_to_travel: float):
        """Move toward target in 2D."""
        if not isinstance(target_point, Point2D):
            raise ValueError("Target must be a Point2D")

        dx = target_point.x - self.x
        dy = target_point.y - self.y

        self.x += dx * prop_to_travel
        self.y += dy * prop_to_travel

    def __repr__(self):
        return f"Center2D({self.x:.2f}, {self.y:.2f})"

# 2D Space with custom distance metrics
class Space2D(Space):
    def __init__(self, min_x=0, max_x=10, min_y=0, max_y=10, metric="euclidean"):
        self.min_x, self.max_x = min_x, max_x
        self.min_y, self.max_y = min_y, max_y
        self.metric = metric

    def distance(self, p1: Point, p2: Point) -> float:
        """Multiple distance metrics."""
        if not isinstance(p1, Point2D) or not isinstance(p2, Point2D):
            raise ValueError("Points must be Point2D instances")

        dx = p1.x - p2.x
        dy = p1.y - p2.y

        if self.metric == "euclidean":
            return math.sqrt(dx*dx + dy*dy)
        elif self.metric == "manhattan": 
            return abs(dx) + abs(dy)
        elif self.metric == "chebyshev":
            return max(abs(dx), abs(dy))
        else:
            raise ValueError(f"Unknown metric: {self.metric}")

    def sample_points(self, n: int) -> list[Point]:
        """Sample random 2D points."""
        points = []
        for _ in range(n):
            x = np.random.uniform(self.min_x, self.max_x)
            y = np.random.uniform(self.min_y, self.max_y)
            points.append(Point2D(self, x, y))
        return points

    def sample_centers(self, k: int) -> list[Center]:
        """Sample random 2D centers."""
        centers = []
        for _ in range(k):
            x = np.random.uniform(self.min_x, self.max_x)
            y = np.random.uniform(self.min_y, self.max_y)
            centers.append(Center2D(self, x, y))
        return centers

    def sample_kpp_centers(self, k: int) -> list[Center]:
        """k-means++ initialization for 2D space."""
        if k <= 0:
            return []

        # Start with a random center
        centers = [Center2D(self, 
                          np.random.uniform(self.min_x, self.max_x),
                          np.random.uniform(self.min_y, self.max_y))]

        # Add remaining centers using k-means++ logic
        for _ in range(k - 1):
            # Sample many candidates and pick the one farthest from existing centers
            best_distance = 0
            best_center = None

            for _ in range(50):  # Try 50 candidates
                candidate = Center2D(self,
                                   np.random.uniform(self.min_x, self.max_x),
                                   np.random.uniform(self.min_y, self.max_y))

                min_dist_to_existing = min(self.distance(candidate, center) for center in centers)

                if min_dist_to_existing > best_distance:
                    best_distance = min_dist_to_existing
                    best_center = candidate

            if best_center is not None:
                centers.append(best_center)

        return centers

    def compute_clusters(self, centers: list[Center]):
        pass

    def calculate_energy_graph(self, centers: list[Center]) -> float:
        return 0.0

    def batch_distances_from_centers(self, centers: list[Center], point: Point) -> list[float]:
        """Compute distances from a point to all centers."""
        return [self.distance(point, center) for center in centers]

# Test different distance metrics

results = {}

for metric in metrics:

    points_2d = space_2d.sample_points(50)

    print(f"{metric.capitalize()} metric - Final energy: {energy_2d:.3f}")

print(f"\nBest metric: {min(results, key=results.get)} (lowest energy)")

    Specialized Example: Circular Space¶

    Here’s an example of clustering on a circle, where distances wrap around:

    # Circular space (angles on a circle)
class CircularPoint(Point):
    def __init__(self, space, angle: float):
        self._space = space  
        self.angle = angle % (2 * math.pi)  # Normalize to [0, 2π)

    @property
    def space(self):
        return self._space

    def __repr__(self):
        return f"CircularPoint(θ={self.angle:.2f})"

class CircularCenter(CircularPoint, Center):
    def brownian_motion(self, time_to_travel: float):
        """Random walk on the circle."""
        step = np.random.normal(0, math.sqrt(time_to_travel))
        self.angle = (self.angle + step) % (2 * math.pi)

    def drift(self, target_point: Point, prop_to_travel: float):
        """Move toward target on circle (shortest path)."""
        if not isinstance(target_point, CircularPoint):
            raise ValueError("Target must be CircularPoint")

        # Calculate shortest angular distance
        diff = target_point.angle - self.angle

        # Wrap to [-π, π] for shortest path
        while diff > math.pi:
            diff -= 2 * math.pi
        while diff < -math.pi:
            diff += 2 * math.pi

        self.angle = (self.angle + diff * prop_to_travel) % (2 * math.pi)

class CircularSpace(Space):
    def distance(self, p1: Point, p2: Point) -> float:
        """Angular distance on circle."""
        if not isinstance(p1, CircularPoint) or not isinstance(p2, CircularPoint):
            raise ValueError("Points must be CircularPoint instances")

        diff = abs(p1.angle - p2.angle)
        return min(diff, 2 * math.pi - diff)  # Shortest arc

    def sample_points(self, n: int) -> list[Point]:
        """Sample random angles."""
        points = []
        for _ in range(n):
            angle = np.random.uniform(0, 2 * math.pi)
            points.append(CircularPoint(self, angle))
        return points

    def sample_centers(self, k: int) -> list[Center]:
        """Sample random center angles."""
        centers = []
        for _ in range(k):
            angle = np.random.uniform(0, 2 * math.pi)
            centers.append(CircularCenter(self, angle))
        return centers

    def sample_kpp_centers(self, k: int) -> list[Center]:
        """k-means++ initialization for circular space."""
        return self.sample_centers(k)  # Simple fallback

    def compute_clusters(self, centers: list[Center]):
        pass

    def calculate_energy_graph(self, centers: list[Center]) -> float:
        return 0.0

    def batch_distances_from_centers(self, centers: list[Center], point: Point) -> list[float]:
        """Compute distances from a point to all centers."""
        return [self.distance(point, center) for center in centers]

# Test circular clustering

    Integration with k-means++ Initialization¶

    For optimal results, implement k-means++ initialization in your Space class:

    # Enhanced 2D space with proper k-means++
class Enhanced2DSpace(Space2D):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self._cached_points = None

    def sample_kpp_centers(self, k: int) -> list[Center]:
        """Proper k-means++ initialization."""
        # Generate a reference set of points for initialization
        if self._cached_points is None:
            self._cached_points = self.sample_points(200)  # Cache for efficiency

        if k <= 0:
            return []

        centers = []

        # Choose first center randomly
        first_point = np.random.choice(self._cached_points)
        centers.append(Center2D(self, first_point.x, first_point.y))

        # Choose remaining centers with probability proportional to squared distance
        for _ in range(k - 1):
            distances = []

            for point in self._cached_points:
                min_dist = min(self.distance(point, center) for center in centers)
                distances.append(min_dist * min_dist)  # Squared distance

            # Weighted random selection
            total = sum(distances)
            if total > 0:
                probs = [d / total for d in distances]
                chosen_idx = np.random.choice(len(self._cached_points), p=probs)
                chosen_point = self._cached_points[chosen_idx]
                centers.append(Center2D(self, chosen_point.x, chosen_point.y))

        return centers

# Test enhanced initialization

points_enhanced = enhanced_space.sample_points(100)

# Compare random vs k-means++ initialization

print(f"k-means++ initialization energy: {energy_kpp:.3f}")
print(f"Improvement: {energy_random - energy_kpp:.3f}")

    class OptimizedSpace2D(Enhanced2DSpace):

    Custom Similarity Spaces¶

    Create spaces based on similarity rather than distance:

    # Similarity-based space (higher values = more similar)
class SimilaritySpace(Space):
    def __init__(self, similarity_matrix: np.ndarray, labels: list = None):
        self.similarity_matrix = similarity_matrix
        self.n_objects = similarity_matrix.shape[0]
        self.labels = labels or list(range(self.n_objects))

        # Convert similarities to distances (distance = max_sim - similarity)
        self.max_sim = np.max(similarity_matrix)
        self.distance_matrix = self.max_sim - similarity_matrix

    def distance(self, p1: Point, p2: Point) -> float:
        """Distance derived from similarity."""
        if not isinstance(p1, SimilarityPoint) or not isinstance(p2, SimilarityPoint):
            raise ValueError("Points must be SimilarityPoint instances")
        return self.distance_matrix[p1.object_id, p2.object_id]

    def sample_points(self, n: int) -> list[Point]:
        """Sample object IDs."""
        points = []
        for _ in range(n):
            obj_id = np.random.randint(0, self.n_objects)
            points.append(SimilarityPoint(self, obj_id))
        return points

    def sample_centers(self, k: int) -> list[Center]:
        """Sample object IDs as centers."""
        centers = []
        chosen_ids = np.random.choice(self.n_objects, size=min(k, self.n_objects), replace=False)
        for obj_id in chosen_ids:
            centers.append(SimilarityCenter(self, obj_id))
        return centers

    def sample_kpp_centers(self, k: int) -> list[Center]:
        """k-means++ initialization for similarity space."""
        return self.sample_centers(k)  # Simple fallback

    def compute_clusters(self, centers: list[Center]):
        pass

    def calculate_energy_graph(self, centers: list[Center]) -> float:
        return 0.0

    def batch_distances_from_centers(self, centers: list[Center], point: Point) -> list[float]:
        """Compute distances from a point to all centers."""
        return [self.distance(point, center) for center in centers]

class SimilarityPoint(Point):
    def __init__(self, space, object_id: int):
        self._space = space
        self.object_id = object_id

    @property
    def space(self):
        return self._space

    def __repr__(self):
        label = self._space.labels[self.object_id] if self._space.labels else self.object_id
        return f"SimilarityPoint({label})"

class SimilarityCenter(SimilarityPoint, Center):
    def brownian_motion(self, time_to_travel: float):
        """Random jump to nearby object."""
        # Find most similar objects and randomly jump to one
        similarities = self._space.similarity_matrix[self.object_id]
        top_similar = np.argsort(similarities)[-5:]  # Top 5 most similar
        self.object_id = np.random.choice(top_similar)

    def drift(self, target_point: Point, prop_to_travel: float):
        """Move toward most similar object to target."""
        if not isinstance(target_point, SimilarityPoint):
            return

        # With some probability, jump to target or similar object
        if np.random.random() < prop_to_travel:
            target_similarities = self._space.similarity_matrix[target_point.object_id]
            most_similar = np.argmax(target_similarities)
            self.object_id = most_similar

    # Performance and correctness checklist
def validate_custom_space(space_class, n_test_points=50):
    """Validate a custom metric space implementation."""
    print(f"Validating {space_class.__name__}...")

    try:
        # Test instantiation
        space = space_class()
        print("✓ Space instantiation successful")

        # Test point sampling
        points = space.sample_points(n_test_points)
        print(f"✓ Point sampling: {len(points)} points generated")

        # Test center sampling  
        centers = space.sample_centers(3)
        print(f"✓ Center sampling: {len(centers)} centers generated")

        # Test distance computation
        if len(points) >= 2:
            dist = space.distance(points[0], points[1])
            print(f"✓ Distance computation: d = {dist:.3f}")

            # Check metric properties
            # 1. Non-negativity
            assert dist >= 0, "Distance must be non-negative"

            # 2. Identity of indiscernibles  
            self_dist = space.distance(points[0], points[0])
            assert abs(self_dist) < 1e-10, "Distance to self must be zero"

            # 3. Symmetry
            reverse_dist = space.distance(points[1], points[0])
            assert abs(dist - reverse_dist) < 1e-10, "Distance must be symmetric"
            print("✓ Metric properties validated")

        # Test clustering integration
        from kmeanssa_ng import SimulatedAnnealing
        sa = SimulatedAnnealing(points[:20], k=2)
        result_centers = sa.run(robust_prop=0.05)
        energy = sa.calculate_energy(result_centers, points[:20])
        print(f"✓ Clustering integration: final energy = {energy:.3f}")

        print(f"✅ {space_class.__name__} validation successful!\n")
        return True

    except Exception as e:
        print(f"❌ {space_class.__name__} validation failed: {e}\n")
        return False
