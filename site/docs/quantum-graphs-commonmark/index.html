    Quantum Graphs¶

    Quantum graphs are the foundation of kmeanssa-ng. This guide provides everything you need to know about working with quantum graphs effectively.

    What Makes Quantum Graphs Special?¶

    Unlike traditional graphs where vertices represent discrete entities, quantum graphs treat edges as first-class citizens:

    • Points can exist anywhere on an edge, not just at vertices
    • Distance is measured along the graph structure (geodesic distance)
    • Each edge has a length attribute representing physical distance
    • The graph must be connected for distance computation
    from kmeanssa_ng import QuantumGraph, QGPoint

# Create a quantum graph manually
graph = QuantumGraph()
graph.add_edge(0, 1, length=2.0)  # Edge with length 2
graph.add_edge(1, 2, length=1.5)  # Edge with length 1.5
graph.add_edge(2, 0, length=3.0)  # Edge with length 3

print(f"Manual graph: {graph.number_of_nodes()} nodes, {graph.number_of_edges()} edges")

    Manual graph: 3 nodes, 3 edges

    # Points can be positioned anywhere on edges
point_a = QGPoint(graph, edge=(0, 1), position=0.25)  # 25% along edge (0,1)
point_b = QGPoint(graph, edge=(1, 2), position=0.80)  # 80% along edge (1,2)

print(f"Point A: {point_a}")
print(f"Point B: {point_b}")

# Distance is computed along the graph
graph.precomputing()  # Essential for distance computation
distance = graph.distance(point_a, point_b)
print(f"Distance: {distance:.3f}")

    Point A: QGPoint on edge (0, 1) at position 0.250
Point B: QGPoint on edge (1, 2) at position 0.800
Distance: 2.550

    Creating Quantum Graphs¶

    Method 1: From Scratch¶

    # Create a more complex graph from scratch
custom_graph = QuantumGraph()

# Add edges with lengths
custom_graph.add_edge('A', 'B', length=1.0)
custom_graph.add_edge('B', 'C', length=2.5)
custom_graph.add_edge('C', 'D', length=1.8)
custom_graph.add_edge('D', 'A', length=2.2)

print(f"Custom graph: {custom_graph.number_of_nodes()} nodes, {custom_graph.number_of_edges()} edges")

# Always validate and precompute
try:
    custom_graph.validate()  # Checks connectivity and edge lengths
    custom_graph.precomputing()  # Caches shortest paths
    print("Graph validation successful!")
except Exception as e:
    print(f"Graph validation failed: {e}")

    Custom graph: 4 nodes, 4 edges
Graph validation failed: 'QuantumGraph' object has no attribute 'validate'

    Method 2: From NetworkX Graphs¶

    import networkx as nx
from kmeanssa_ng import as_quantum_graph

# Start with any NetworkX graph
G = nx.erdos_renyi_graph(20, 0.3)
print(f"NetworkX graph: {G.number_of_nodes()} nodes, {G.number_of_edges()} edges")

# Convert with uniform edge lengths
quantum_graph_uniform = as_quantum_graph(G, edge_length=1.0)
quantum_graph_uniform.precomputing()
print(f"Converted quantum graph: {quantum_graph_uniform.number_of_nodes()} nodes")

    NetworkX graph: 20 nodes, 60 edges
Converted quantum graph: 20 nodes

    # Or use existing edge attributes with custom weighting
G_custom = nx.karate_club_graph()
print(f"Karate club graph: {G_custom.number_of_nodes()} nodes, {G_custom.number_of_edges()} edges")

# Convert to quantum graph first
karate_quantum = as_quantum_graph(G_custom, edge_length=1.0)

# Add custom lengths based on some property
for (u, v) in karate_quantum.edges():
    custom_length = 1.0 + (u + v) * 0.1  # Example weighting
    karate_quantum[u][v]['length'] = custom_length

karate_quantum.precomputing()
print(f"Custom weighted karate graph created")

    Karate club graph: 34 nodes, 78 edges
Custom weighted karate graph created

    Working with Points and Centers¶

    QGPoint - Data Points¶

    Points represent data locations on the quantum graph:

    # Create points at specific locations using the SBM graph
# First, let's see what edges are available
available_edges = list(sbm_graph.edges())[:5]  # Get first 5 edges
print(f"Some available edges: {available_edges}")

# Create points using existing edges
point1 = QGPoint(sbm_graph, edge=available_edges[0], position=0.3)
point2 = QGPoint(sbm_graph, edge=available_edges[1], position=0.7)

# Points are immutable and hashable
print(f"Point 1: {point1}")
print(f"Point 2: {point2}")
print(f"Hash of point1: {hash(point1)}")  # Can be used in sets/dicts

# Access properties
print(f"Point1 edge: {point1.edge}")
print(f"Point1 position: {point1.position}")

    QGCenter - Cluster Centers¶

    Centers can move during the clustering process:

    from kmeanssa_ng import QGCenter

# Create a center from a QGPoint using an existing edge
simple_edges = list(simple_graph.edges())[:3]
print(f"Simple graph edges: {simple_edges}")

initial_point = QGPoint(simple_graph, edge=simple_edges[0], position=0.5)
center = QGCenter(initial_point)
print(f"Initial center: {center}")

# Centers can perform operations (these modify the center in place)
h = 0.1  # Step size for Brownian motion
center.brownian_motion(h)
print(f"After Brownian motion: {center}")

# Get the closest node (useful for interpretation)
closest_node = center._closest_node()
print(f"Center closest to node: {closest_node}")

    Simple graph edges: [('A0', 'B0'), ('A0', 'A1'), ('A0', 'A2')]
Initial center: QGPoint on edge ('A0', 'B0') at position 0.500
After Brownian motion: QGPoint on edge ('A0', 'B0') at position 0.557
Center closest to node: A0

    Distance Computation¶

    Understanding how distances are computed is crucial for effective use:

    Same Edge Distance¶

    # Two points on the same edge
first_edge = available_edges[0]
same_edge_point1 = QGPoint(sbm_graph, edge=first_edge, position=0.2)  # 20% along edge
same_edge_point2 = QGPoint(sbm_graph, edge=first_edge, position=0.8)  # 80% along edge

# Distance = |0.8 - 0.2| * edge_length
same_edge_distance = sbm_graph.distance(same_edge_point1, same_edge_point2)
edge_length = sbm_graph[first_edge[0]][first_edge[1]]['length']
expected_distance = abs(0.8 - 0.2) * edge_length

print(f"Same edge distance: {same_edge_distance:.3f}")
print(f"Expected distance: {expected_distance:.3f}")
print(f"Edge length: {edge_length:.3f}")

    Same edge distance: 0.600
Expected distance: 0.600
Edge length: 1.000

    Different Edge Distance¶

    For points on different edges, the algorithm finds the shortest path:

    # Points on different edges
diff_edge_point1 = QGPoint(sbm_graph, edge=available_edges[0], position=0.3)
diff_edge_point2 = QGPoint(sbm_graph, edge=available_edges[2], position=0.7)

# Complex geodesic distance computation
diff_edge_distance = sbm_graph.distance(diff_edge_point1, diff_edge_point2)
print(f"Different edges distance: {diff_edge_distance:.3f}")

# This involves finding shortest paths between nodes and combining distances

    Different edges distance: 1.000

    Performance: Precomputing¶

    Always call precomputing() before intensive distance calculations:

    import time
from kmeanssa_ng import generate_sbm

# Create a test graph for timing
timing_graph = generate_sbm(sizes=[15, 15], p=[[0.8, 0.1], [0.1, 0.8]])
timing_points = timing_graph.sample_points(20)

# Without precomputing (slower)
start_time = time.time()
distance_no_cache = timing_graph.distance(timing_points[0], timing_points[1])
time_no_cache = time.time() - start_time

# With precomputing (faster)
timing_graph.precomputing()
start_time = time.time()
distance_with_cache = timing_graph.distance(timing_points[0], timing_points[1])
time_with_cache = time.time() - start_time

print(f"Without precomputing: {distance_no_cache:.3f} (time: {time_no_cache:.6f}s)")
print(f"With precomputing: {distance_with_cache:.3f} (time: {time_with_cache:.6f}s)")
speedup = time_no_cache / time_with_cache if time_with_cache > 0 else float('inf')
print(f"Speedup: {speedup:.1f}x faster")

    Without precomputing: 0.000 (time: 0.000052s)
With precomputing: 0.000 (time: 0.000035s)
Speedup: 1.5x faster

    Custom Sampling (Advanced)¶

    # If you need custom sampling, implement it manually
import random

def custom_sample_points(graph, n_points):
    points = []
    edges = list(graph.edges())

    for _ in range(n_points):
        # Choose edge (you can customize this distribution)
        edge = random.choice(edges)

        # Choose position on edge
        position = random.random()

        points.append(QGPoint(graph, edge=edge, position=position))

    return points

custom_points = custom_sample_points(simple_graph, 20)
print(f"Custom sampled {len(custom_points)} points")
print(f"Sample custom points: {custom_points[:3]}")

    # Check connectivity of our graphs
graphs_to_check = [("SBM", sbm_graph), ("Simple", simple_graph), ("Custom", custom_graph)]

    Clustering with Quantum Graph Features¶

    Using QGSimulatedAnnealing¶

    from kmeanssa_ng import QGSimulatedAnnealing, SimulatedAnnealing

# Sample points from SBM graph for clustering
clustering_points = sbm_graph.sample_points(100)

# Use quantum graph specific clustering
qg_sa = QGSimulatedAnnealing(clustering_points, k=3)
node_centers = qg_sa.run_for_kmeans(robust_prop=0.1)
print(f"QG centers near nodes: {node_centers}")

# Compare with exact positions
regular_sa = SimulatedAnnealing(clustering_points, k=3)
exact_centers = regular_sa.run(initialization="kpp")
print(f"Found {len(exact_centers)} exact centers")

    QG centers near nodes: [55, 43, 19]
Found 3 exact centers

    Multi-Scale Analysis¶

    # Analyze clustering at different scales
print("Multi-scale clustering analysis:")

    Real-World Applications¶

    Transportation Networks¶

    # Model a city road network
city_graph = QuantumGraph()

# Add roads with travel times as lengths (in minutes)
roads = [
    ('MainSt_1st', 'MainSt_2nd', 2.5),
    ('MainSt_2nd', 'MainSt_3rd', 3.0),
    ('MainSt_3rd', 'Oak_Ave', 1.8),
    ('Oak_Ave', 'MainSt_1st', 4.2)
]

for start, end, travel_time in roads:
    city_graph.add_edge(start, end, length=travel_time)

city_graph.precomputing()

# Points represent locations (GPS coordinates mapped to roads)
restaurant = QGPoint(city_graph, edge=('MainSt_1st', 'MainSt_2nd'), position=0.3)
school = QGPoint(city_graph, edge=('MainSt_2nd', 'MainSt_3rd'), position=0.8)

travel_time = city_graph.distance(restaurant, school)
print(f"Travel time from restaurant to school: {travel_time:.2f} minutes")

    Travel time from restaurant to school: 3.00 minutes

    Performance Optimization¶

    # Optimize for large graphs
large_graph = generate_sbm(sizes=[50, 50], p=[[0.7, 0.1], [0.1, 0.7]])
large_graph.precomputing()  # Essential!

# Limit point sampling for performance
n_points = min(200, len(large_graph.edges()) * 5)
large_points = large_graph.sample_points(n_points)

print(f"Large graph: {large_graph.number_of_nodes()} nodes, {len(large_points)} points")

# Use quantum graph clustering with optimized parameters
large_qg_sa = QGSimulatedAnnealing(large_points, k=2, step_size=0.2)
large_centers = large_qg_sa.run_for_kmeans(robust_prop=0.05)
print(f"Large graph clustering centers: {large_centers}")

    Large graph: 100 nodes, 200 points
Large graph clustering centers: [55, 15]
