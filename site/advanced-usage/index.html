    def adaptive_parameter_selection(points, k, max_iterations=10):
    """Adaptively select parameters based on graph properties."""

    # Start with default parameters
    current_params = {'lambda_param': 1.0, 'beta': 1.0, 'step_size': 0.1}
    current_energy = float('inf')

    # Graph analysis for parameter hints
    n_points = len(points)
    graph = points[0].space

    # Estimate graph density and complexity
    n_nodes = len(graph.nodes())
    n_edges = len(graph.edges())
    density = 2 * n_edges / (n_nodes * (n_nodes - 1)) if n_nodes > 1 else 0

    print(f"Graph analysis: {n_nodes} nodes, density={density:.3f}")

    # Adaptive rules
    if density > 0.1:  # Dense graph
        current_params['lambda_param'] = 2.0  # Higher exploration
        current_params['step_size'] = 0.05    # Smaller steps
        print("Dense graph detected → increased exploration, smaller steps")
    elif density < 0.05:  # Sparse graph
        current_params['beta'] = 2.0          # Faster convergence
        current_params['step_size'] = 0.15    # Larger steps
        print("Sparse graph detected → faster convergence, larger steps")

    if n_points > 200:  # Many points
        current_params['lambda_param'] *= 1.5
        print(f"Many points ({n_points}) → increased exploration")

    # Test adaptive parameters
    sa_adaptive = SimulatedAnnealing(points, k=k, **current_params)
    centers_adaptive = sa_adaptive.run(robust_prop=0.1, initialization="kpp")
    adaptive_energy = sa_adaptive.calculate_energy(centers_adaptive, points)

    print(f"Adaptive parameters: {current_params}")
    print(f"Adaptive energy: {adaptive_energy:.3f}")

    return centers_adaptive, current_params

    Graph Cleaning Techniques¶

    def clean_graph_by_centrality(graph, threshold=0.1):
    """Remove edges with low centrality to reduce noise."""
    # Calculate edge betweenness centrality
    edge_centrality = nx.edge_betweenness_centrality(graph, weight='length')

    # Identify edges to remove
    edges_to_remove = [
        edge for edge, centrality in edge_centrality.items() 
        if centrality < threshold
    ]

    print(f"Removing {len(edges_to_remove)} edges with centrality < {threshold}")

    # Create cleaned graph
    cleaned_graph = graph.copy()
    cleaned_graph.remove_edges_from(edges_to_remove)

    # Ensure connectivity
    if not nx.is_connected(cleaned_graph):
        # Work with largest component
        largest_cc = max(nx.connected_components(cleaned_graph), key=len)
        cleaned_graph = cleaned_graph.subgraph(largest_cc).copy()
        print(f"Keeping largest component: {len(cleaned_graph.nodes())} nodes")

    return cleaned_graph

# Apply cleaning with less aggressive threshold
cleaned_graph = clean_graph_by_centrality(noisy_graph, threshold=0.01)

if len(cleaned_graph.nodes()) > 10 and len(cleaned_graph.edges()) > 5:

    Clustering with Size Constraints¶

    def constrained_clustering(points, k, min_cluster_size=None, max_cluster_size=None):
    """Clustering with cluster size constraints."""

    if min_cluster_size is None:
        min_cluster_size = len(points) // (k * 3)  # At least 1/3 of average
    if max_cluster_size is None:
        max_cluster_size = len(points) // k * 2    # At most 2x average

    print(f"Constrained clustering: {min_cluster_size} ≤ cluster size ≤ {max_cluster_size}")

    best_centers = None
    best_score = float('inf')
    attempts = 0
    max_attempts = 20

    while attempts < max_attempts:
        # Run standard clustering
        sa = SimulatedAnnealing(points, k=k)
        centers = sa.run(initialization="kpp", robust_prop=0.05)

        # Assign points to clusters
        cluster_assignments = []
        cluster_sizes = [0] * k

        for point in points:
            distances = [point.space.distance(point, center) for center in centers]
            closest_cluster = np.argmin(distances)
            cluster_assignments.append(closest_cluster)
            cluster_sizes[closest_cluster] += 1

        # Check size constraints
        valid_clustering = all(
            min_cluster_size <= size <= max_cluster_size 
            for size in cluster_sizes
        )

        if valid_clustering:
            energy = sa.calculate_energy(centers, points)
            if energy < best_score:
                best_score = energy
                best_centers = centers
                print(f"Found valid clustering: energy={energy:.3f}, sizes={cluster_sizes}")
                break

        attempts += 1

    if best_centers is None:
        print(f"Warning: Could not find valid clustering in {max_attempts} attempts")
        # Fallback to unconstrained
        sa = SimulatedAnnealing(points, k=k)
        best_centers = sa.run(initialization="kpp", robust_prop=0.1)

    return best_centers

    Multi-Objective Optimization¶

    def multi_objective_clustering(points, k, objectives=['energy', 'silhouette', 'separation']):
    """Multi-objective clustering optimization."""

    def calculate_silhouette_score(points, centers):
        """Simple silhouette score approximation."""
        if len(centers) < 2:
            return 0

        scores = []
        for point in points[:20]:  # Sample for speed
            distances = [point.space.distance(point, center) for center in centers]
            closest_idx = np.argmin(distances)

            # Average distance to own cluster center (approximation)
            a = distances[closest_idx]

            # Average distance to nearest other cluster center
            other_distances = [distances[i] for i in range(len(centers)) if i != closest_idx]
            b = min(other_distances) if other_distances else a

            if b > a:
                scores.append((b - a) / b)
            else:
                scores.append(0)

        return np.mean(scores) if scores else 0

    def calculate_separation(centers):
        """Average separation between cluster centers."""
        if len(centers) < 2:
            return 0

        separations = []
        for i, center1 in enumerate(centers):
            for center2 in centers[i+1:]:
                separations.append(center1.space.distance(center1, center2))

        return np.mean(separations)

    # Multiple runs with different trade-offs
    results = []
    weights_sets = [
        {'energy': 1.0, 'silhouette': 0.0, 'separation': 0.0},    # Pure energy
        {'energy': 0.7, 'silhouette': 0.3, 'separation': 0.0},   # Energy + silhouette
        {'energy': 0.5, 'silhouette': 0.3, 'separation': 0.2},   # Balanced
        {'energy': 0.4, 'silhouette': 0.4, 'separation': 0.2},   # Silhouette focused
    ]

    for i, weights in enumerate(weights_sets):
        sa = SimulatedAnnealing(points, k=k)
        centers = sa.run(initialization="kpp", robust_prop=0.1)

        # Calculate all objectives
        energy = sa.calculate_energy(centers, points)
        silhouette = calculate_silhouette_score(points, centers)
        separation = calculate_separation(centers)

        # Normalize and combine (simplified)
        normalized_energy = energy / len(points)  # Rough normalization
        combined_score = (weights['energy'] * normalized_energy - 
                         weights['silhouette'] * silhouette + 
                         weights['separation'] / separation if separation > 0 else 0)

        results.append({
            'weights': weights,
            'centers': centers,
            'energy': energy,
            'silhouette': silhouette,
            'separation': separation,
            'combined_score': combined_score
        })

        print(f"Config {i+1}: Energy={energy:.3f}, Silhouette={silhouette:.3f}, Separation={separation:.3f}")

    # Find best according to combined score
    best_result = min(results, key=lambda x: x['combined_score'])
    print(f"\nBest multi-objective solution:")
    print(f"Weights: {best_result['weights']}")
    print(f"Energy: {best_result['energy']:.3f}")
    print(f"Silhouette: {best_result['silhouette']:.3f}")
    print(f"Separation: {best_result['separation']:.3f}")

    return best_result['centers']

    def analyze_clustering_stability(points, k, n_runs=10):
    """Analyze stability of clustering across multiple runs."""

    all_energies = []
    all_center_positions = []

    print(f"Analyzing clustering stability over {n_runs} runs...")

    for run in range(n_runs):
        sa = SimulatedAnnealing(points, k=k)
        centers = sa.run(initialization="kpp", robust_prop=0.1)
        energy = sa.calculate_energy(centers, points)

        all_energies.append(energy)

        # Extract center positions (for quantum graphs, get representative positions)
        if hasattr(centers[0], 'edge') and hasattr(centers[0], 'position'):
            center_positions = [(c.edge, c.position) for c in centers]
        else:
            center_positions = [str(c) for c in centers]  # Fallback representation

        all_center_positions.append(center_positions)

        if run % 3 == 0:
            print(f"Run {run+1}/{n_runs}: Energy = {energy:.3f}")

    # Stability analysis
    energy_mean = np.mean(all_energies)
    energy_std = np.std(all_energies)
    energy_cv = energy_std / energy_mean if energy_mean > 0 else 0

    print(f"\nStability Analysis:")
    print(f"Energy: μ={energy_mean:.3f}, σ={energy_std:.3f}, CV={energy_cv:.3f}")
    print(f"Best energy: {min(all_energies):.3f}")
    print(f"Worst energy: {max(all_energies):.3f}")

    # Stability quality assessment
    if energy_cv < 0.05:
        stability = "Excellent"
    elif energy_cv < 0.10:
        stability = "Good"  
    elif energy_cv < 0.20:
        stability = "Fair"
    else:
        stability = "Poor"

    print(f"Stability assessment: {stability}")

    return {
        'energies': all_energies,
        'mean_energy': energy_mean,
        'std_energy': energy_std,
        'stability': stability
    }

    import time
from contextlib import contextmanager

@contextmanager
def timer(description):
    """Context manager for timing operations."""
    start = time.time()
    yield
    end = time.time()
    print(f"{description}: {end - start:.3f} seconds")

def performance_profile(points, k_values=[2, 3, 4, 5]):
    """Profile performance across different k values and parameters."""

    print(f"Performance profiling with {len(points)} points...")

    results = {}

    for k in k_values:
        print(f"\n--- k = {k} ---")

        # Test different algorithms and parameters
        configs = [
            ('Standard v1', {'algorithm_version': 'v1', 'robust_prop': 0.1}),
            ('Standard v2', {'algorithm_version': 'v2', 'robust_prop': 0.1}),
            ('Fast', {'robust_prop': 0.05, 'beta': 2.0, 'step_size': 0.2}),
            ('Precise', {'robust_prop': 0.15, 'beta': 0.5, 'step_size': 0.05}),
        ]

        k_results = {}

        for name, params in configs:
            with timer(f"{name:>12}"):
                sa = SimulatedAnnealing(points, k=k, **{p: v for p, v in params.items() 
                                                       if p not in ['algorithm_version', 'robust_prop']})
                algorithm_version = params.get('algorithm_version', 'v1')
                robust_prop = params.get('robust_prop', 0.1)

                centers = sa.run(
                    algorithm_version=algorithm_version,
                    robust_prop=robust_prop,
                    initialization="kpp"
                )
                energy = sa.calculate_energy(centers, points)

            k_results[name] = {'energy': energy, 'centers': centers}
            print(f"             Energy: {energy:.3f}")

        results[k] = k_results

    # Summary of best configurations
    print(f"\n=== Performance Summary ===")
    for k, k_results in results.items():
        best_config = min(k_results.items(), key=lambda x: x[1]['energy'])
        print(f"k={k}: Best = {best_config[0]} (energy={best_config[1]['energy']:.3f})")

    return results

    def export_clustering_results(centers, points, filename_base="clustering_results"):
    """Export clustering results in multiple formats."""

    # Assign points to clusters
    assignments = []
    for point in points:
        distances = [point.space.distance(point, center) for center in centers]
        assignments.append(np.argmin(distances))

    # Create results dictionary
    results = {
        'n_points': len(points),
        'n_clusters': len(centers),
        'cluster_assignments': assignments,
        'cluster_sizes': [assignments.count(i) for i in range(len(centers))]
    }

    # Export basic statistics
    print(f"Clustering Results Export:")
    print(f"Points: {results['n_points']}")
    print(f"Clusters: {results['n_clusters']}")
    print(f"Cluster sizes: {results['cluster_sizes']}")

    # Calculate clustering quality metrics
    total_energy = sum(
        min(point.space.distance(point, center)**2 for center in centers)
        for point in points
    )
    avg_energy = total_energy / len(points)

    results['total_energy'] = total_energy
    results['average_energy'] = avg_energy

    print(f"Average energy per point: {avg_energy:.3f}")

    return results

def compare_with_baseline(points, k, baseline_method="random"):
    """Compare clustering results with baseline methods."""

    # Our method
    sa = SimulatedAnnealing(points, k=k)
    our_centers = sa.run(initialization="kpp", robust_prop=0.1)
    our_energy = sa.calculate_energy(our_centers, points)

    # Baseline methods
    baseline_results = {}

    if baseline_method == "random":
        # Random clustering
        random_centers = points[0].space.sample_centers(k)
        random_energy = sa.calculate_energy(random_centers, points)
        baseline_results['random'] = random_energy

    # K-means++ initialization only (no optimization)
    kpp_centers = points[0].space.sample_kpp_centers(k)
    kpp_energy = sa.calculate_energy(kpp_centers, points)
    baseline_results['kpp_only'] = kpp_energy

    print(f"Method Comparison (lower is better):")
    print(f"Our SA method:     {our_energy:.3f}")
    for method, energy in baseline_results.items():
        improvement = energy - our_energy
        improvement_pct = (improvement / energy) * 100 if energy > 0 else 0
        print(f"{method:>15}: {energy:.3f} (improvement: {improvement:+.3f}, {improvement_pct:.1f}%)")

    return our_centers, baseline_results
