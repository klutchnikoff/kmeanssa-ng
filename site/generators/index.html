    Graph Generators¶

    Graph generators are essential for testing, benchmarking, and experimenting with clustering algorithms. This guide covers all the graph generation functions in kmeanssa-ng, from simple test cases to complex stochastic models.

    Overview¶

    The library provides several types of graph generators:

    • Simple graphs: Two-cluster test cases for development
    • Stochastic Block Models (SBM): Multi-cluster graphs with controllable structure
    • NetworkX converters: Transform existing graphs to quantum graphs
    • Complete graphs: Dense graphs from similarity matrices
    import numpy as np
import networkx as nx
from kmeanssa_ng import (
    generate_simple_graph, 
    generate_sbm, 
    generate_random_sbm,
    as_quantum_graph,
    complete_quantum_graph
)

print("Graph generators loaded successfully!")

    Graph generators loaded successfully!

    Simple Graphs¶

    Basic Two-Cluster Graph¶

    The simplest generator creates symmetric two-cluster graphs connected by a bridge:

    # Create a basic two-cluster graph
simple_graph = generate_simple_graph(
    n_a=5,              # 5 neighbors for each central node
    n_aa=3,             # 3 second-level neighbors for each first-level node  
    bridge_length=2.0   # Distance between clusters
)

print(f"Simple graph: {simple_graph.number_of_nodes()} nodes, {simple_graph.number_of_edges()} edges")
print(f"Cluster A central node: A0")
print(f"Cluster B central node: B0")
print(f"Bridge length: {simple_graph.get_edge_data('A0', 'B0')['length']}")

# Graph comes pre-computed and ready to use
points = simple_graph.sample_points(50)
print(f"Sampled {len(points)} points for clustering")

    Simple graph: 42 nodes, 41 edges
Cluster A central node: A0
Cluster B central node: B0
Bridge length: 2.0
Sampled 50 points for clustering

    Stochastic Block Models¶

    Basic SBM¶

    Stochastic Block Models create multi-cluster graphs with controllable intra- and inter-cluster connectivity:

    # Create a two-block SBM
sbm_graph = generate_sbm(
    sizes=[40, 60],                    # 40 nodes in block 1, 60 in block 2
    p=[[0.8, 0.1], [0.1, 0.7]]       # High intra-block, low inter-block probability
)

print(f"SBM graph: {sbm_graph.number_of_nodes()} nodes, {sbm_graph.number_of_edges()} edges")
print(f"Expected edges within block 1: ~{40*39/2 * 0.8:.0f}")
print(f"Expected edges within block 2: ~{60*59/2 * 0.7:.0f}")  
print(f"Expected edges between blocks: ~{40*60 * 0.1:.0f}")

    SBM graph: 100 nodes, 2086 edges
Expected edges within block 1: ~624
Expected edges within block 2: ~1239
Expected edges between blocks: ~240

    Multi-Block SBM¶

    # Create a three-block SBM with different cluster strengths
three_block_sbm = generate_sbm(
    sizes=[30, 25, 35],
    p=[[0.9, 0.05, 0.02],   # Block 1: very tight
       [0.05, 0.7, 0.15],    # Block 2: medium, some connection to block 3
       [0.02, 0.15, 0.8]]    # Block 3: tight, connected to block 2
)

print(f"Three-block SBM: {three_block_sbm.number_of_nodes()} nodes")

# Check block assignments
block_attr = nx.get_node_attributes(three_block_sbm, 'block')
block_counts = {}
for node, block in block_attr.items():
    block_counts[block] = block_counts.get(block, 0) + 1

print(f"Block sizes: {dict(sorted(block_counts.items()))}")

    Three-block SBM: 90 nodes
Block sizes: {0: 30, 1: 25, 2: 35}

    With Edge Attributes¶

    # Use existing edge attributes as lengths
weighted_graph = nx.Graph()
weighted_graph.add_edge(0, 1, weight=0.5)
weighted_graph.add_edge(1, 2, weight=1.2)
weighted_graph.add_edge(2, 3, weight=0.8)
weighted_graph.add_edge(3, 0, weight=2.0)

# Convert using existing weights as lengths
weighted_qg = as_quantum_graph(weighted_graph, edge_length=1.0)  # Default for missing weights

# Manually set lengths from weights
for u, v, data in weighted_graph.edges(data=True):
    if 'weight' in data:
        weighted_qg[u][v]['length'] = data['weight']

weighted_qg.validate_edge_lengths()
weighted_qg.precomputing()

print(f"Weighted graph converted with edge lengths: {[d['length'] for u,v,d in weighted_qg.edges(data=True)]}")

    Weighted graph converted with edge lengths: [0.5, 2.0, 1.2, 0.8]

    Custom Node and Edge Weights¶

    # Create graph with custom sampling weights
custom_graph = generate_simple_graph(n_a=6, n_aa=3, bridge_length=4.0)

# Modify node weights for different sampling probabilities
node_weights = {}
for node in custom_graph.nodes():
    if node.endswith('0'):  # Central nodes
        node_weights[node] = 5.0
    elif len(node) == 2:   # First-level nodes  
        node_weights[node] = 3.0
    else:                  # Second-level nodes
        node_weights[node] = 1.0

nx.set_node_attributes(custom_graph, node_weights, 'weight')

# Sample points (more likely from high-weight nodes)
weighted_points = custom_graph.sample_points(60)
print(f"Sampled {len(weighted_points)} points with custom weights")

# Check sampling distribution
point_nodes = [p.edge[0] if p.position < 0.5 else p.edge[1] for p in weighted_points[:20]]
central_count = sum(1 for node in point_nodes if str(node).endswith('0'))
print(f"Points near central nodes: {central_count}/20 (should be high due to weight=5)")

    Sampled 60 points with custom weights
Points near central nodes: 1/20 (should be high due to weight=5)

    Graph Modification and Testing¶

    # Start with SBM and modify structure
base_graph = generate_sbm(sizes=[30, 30], p=[[0.7, 0.1], [0.1, 0.7]])

# Add extra inter-cluster connections
block_attr = nx.get_node_attributes(base_graph, 'block')
block_0_nodes = [n for n, b in block_attr.items() if b == 0]
block_1_nodes = [n for n, b in block_attr.items() if b == 1] 

# Add 5 random connections between blocks
import random
for _ in range(5):
    n1 = random.choice(block_0_nodes)
    n2 = random.choice(block_1_nodes)
    if not base_graph.has_edge(n1, n2):
        base_graph.add_edge(n1, n2, length=1.0, weight=1.0)

print(f"Modified SBM: {base_graph.number_of_edges()} edges")
base_graph.precomputing()

# Test clustering on modified graph
modified_points = base_graph.sample_points(100)

modified_centers = sa_modified.run(initialization="kpp")

    Graph Validation¶

    # Validate generated graphs
def validate_quantum_graph(graph, name):
    """Validate that a quantum graph is properly constructed."""
    issues = []

    # Check connectivity
    if not nx.is_connected(graph):
        issues.append("Graph not connected")

    # Check edge lengths
    try:
        graph.validate_edge_lengths()
    except ValueError as e:
        issues.append(f"Edge validation failed: {e}")

    # Check node weights
    node_weights = nx.get_node_attributes(graph, 'weight')
    if not all(w > 0 for w in node_weights.values()):
        issues.append("Some node weights are not positive")

    # Check if precomputing works
    try:
        graph.precomputing() 
    except Exception as e:
        issues.append(f"Precomputing failed: {e}")

    if issues:
        print(f"⚠️  {name}: {', '.join(issues)}")
    else:
        print(f"✅ {name}: Valid quantum graph")

# Test all generator types
test_graphs = [
    ("Simple graph", generate_simple_graph()),
    ("SBM graph", generate_sbm()),
    ("Random SBM", generate_random_sbm()),
    ("Converted NetworkX", as_quantum_graph(nx.path_graph(5)))
]

for name, graph in test_graphs:
    validate_quantum_graph(graph, name)

    ✅ Simple graph: Valid quantum graph
✅ SBM graph: Valid quantum graph
✅ Random SBM: Valid quantum graph
✅ Converted NetworkX: Valid quantum graph

    Best Practices¶

    Choosing the Right Generator¶

    # Guidelines for generator selection
guidelines = {
    "Development/Testing": "generate_simple_graph() - predictable structure",
    "Algorithm benchmarking": "generate_sbm() - controlled clustering difficulty", 
    "Real-world modeling": "as_quantum_graph() - convert domain-specific graphs",
    "Dense clustering": "complete_quantum_graph() - all pairwise distances",
    "Variable cluster sizes": "generate_random_sbm() - asymmetric blocks"
}

print("Generator selection guidelines:")
for use_case, recommendation in guidelines.items():
    print(f"• {use_case}: {recommendation}")

    Generator selection guidelines:
• Development/Testing: generate_simple_graph() - predictable structure
• Algorithm benchmarking: generate_sbm() - controlled clustering difficulty
• Real-world modeling: as_quantum_graph() - convert domain-specific graphs
• Dense clustering: complete_quantum_graph() - all pairwise distances
• Variable cluster sizes: generate_random_sbm() - asymmetric blocks

    Performance Considerations¶

    # Performance tips for large graphs
performance_tips = [
    "Call precomputing() immediately after generation",
    "Use connected graphs to avoid validation issues", 
    "Limit point sampling for very large graphs (>1000 nodes)",
    "Consider edge weight distributions for realistic sampling",
    "Test with small graphs first, then scale up"

print("\nPerformance tips:")
for i, tip in enumerate(performance_tips, 1):
    print(f"{i}. {tip}")

    Performance tips:
1. Call precomputing() immediately after generation
2. Use connected graphs to avoid validation issues
3. Limit point sampling for very large graphs (>1000 nodes)
4. Consider edge weight distributions for realistic sampling
5. Test with small graphs first, then scale up
