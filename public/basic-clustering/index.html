    Basic Clustering Examples¶

    This guide provides practical examples of using kmeanssa-ng for various clustering scenarios.

    Example 1: Two-Cluster Stochastic Block Model¶

    The most basic scenario - clustering points on a graph with clear community structure:

    from kmeanssa_ng import generate_sbm, QGSimulatedAnnealing, SimulatedAnnealing

# Generate a graph with two distinct communities
graph = generate_sbm(
    sizes=[40, 40],       # Two communities of 40 nodes each
    p=[[0.8, 0.1],        # High intra-community connectivity  
       [0.1, 0.8]],       # Low inter-community connectivity
)

# Essential: precompute shortest paths
graph.precomputing()

# Sample points uniformly across the graph
points = graph.sample_points(150)

print(f"Created graph with {graph.number_of_nodes()} nodes and {graph.number_of_edges()} edges")
print(f"Sampled {len(points)} points on the graph")

    Created graph with 80 nodes and 1377 edges
Sampled 150 points on the graph

    # Run quantum graph specialized simulated annealing
qg_sa = QGSimulatedAnnealing(
    observations=points,
    k=2,                  # We know there are 2 clusters
    lambda_param=1.0,     # Standard temperature
    beta=1.0,            # Standard drift strength
    step_size=0.1        # Standard step size
)

# Get cluster centers as node IDs (more interpretable)
node_centers = qg_sa.run_for_kmeans(robust_prop=0.1)
print(f"Cluster centers near nodes: {node_centers}")

# For comparison, exact positions with base class
sa = SimulatedAnnealing(points, k=2)
centers = sa.run(robust_prop=0.1, initialization="kpp")
print(f"Found {len(centers)} exact centers")

    Cluster centers near nodes: [44, 12]
Found 2 exact centers

    # Display results
print("\nQuantum Graph specific results:")
for i, node_id in enumerate(node_centers):
    print(f"  Cluster {i+1} center near node: {node_id}")

print("\nExact center positions:")
for i, center in enumerate(centers[:2]):  # Show first 2
    print(f"  Cluster {i+1}: {center}")

    Quantum Graph specific results:
  Cluster 1 center near node: 44
  Cluster 2 center near node: 12

Exact center positions:
  Cluster 1: QGPoint on 'stochastic_block_model' edge (69, 39) at position -0.278
  Cluster 2: QGPoint on 'stochastic_block_model' edge (26, 12) at position 0.373

    Example 2: Multi-Cluster Complex Graph¶

    Handling more complex scenarios with multiple clusters:

    # Create a 4-cluster graph with varying connectivity
graph_4 = generate_sbm(
    sizes=[20, 30, 25, 35],  # Different cluster sizes
    p=[[0.9, 0.1, 0.05, 0.02],  # Strong within-cluster connectivity
       [0.1, 0.8, 0.15, 0.05],  # Moderate cross-cluster connections
       [0.05, 0.15, 0.85, 0.1], 
       [0.02, 0.05, 0.1, 0.9]]
)

graph_4.precomputing()
points_4 = graph_4.sample_points(200)

print(f"4-cluster graph: {graph_4.number_of_nodes()} nodes, {graph_4.number_of_edges()} edges")

    4-cluster graph: 110 nodes, 1684 edges

    # For more clusters, you might need more exploration
qg_sa_4 = QGSimulatedAnnealing(
    observations=points_4,
    k=4,
    lambda_param=1.5,     # Slightly higher temperature for exploration
    beta=0.8,            # Slightly lower drift for broader search
    step_size=0.15       # Slightly larger steps
)

node_centers_4 = qg_sa_4.run_for_kmeans(robust_prop=0.15)
print(f"4-cluster result: {len(node_centers_4)} centers near nodes: {node_centers_4}")

    4-cluster result: 4 centers near nodes: [75, 1, 34, 38]

    Example 3: Using Existing NetworkX Graphs¶

    Convert real-world networks to quantum graphs:

    import networkx as nx
from kmeanssa_ng import as_quantum_graph

# Start with a well-known network
G = nx.karate_club_graph()

# Convert to quantum graph with uniform edge lengths
karate_graph = as_quantum_graph(G, edge_length=1.0)
karate_graph.precomputing()

print(f"Karate club graph: {karate_graph.number_of_nodes()} nodes, {karate_graph.number_of_edges()} edges")

    Karate club graph: 34 nodes, 78 edges

    # Sample points and cluster
karate_points = karate_graph.sample_points(50)
qg_sa_karate = QGSimulatedAnnealing(karate_points, k=2)  # Karate club has 2 known communities
karate_nodes = qg_sa_karate.run_for_kmeans(robust_prop=0.1)

print(f"Karate club clustering centers near nodes: {karate_nodes}")

    Karate club clustering centers near nodes: [0, 33]

    # You can also use custom edge lengths based on graph properties
G_weighted = nx.karate_club_graph()

# Add custom weights (e.g., inverse of edge betweenness)
betweenness = nx.edge_betweenness_centrality(G_weighted)
for (u, v), centrality in betweenness.items():
    G_weighted[u][v]['length'] = 1.0 / (centrality + 0.01)  # Avoid division by zero

# Convert to quantum graph first, then update lengths
karate_weighted = as_quantum_graph(G_weighted, edge_length=1.0)  # Default lengths
# Update with custom lengths
for (u, v), centrality in betweenness.items():
    karate_weighted[u][v]['length'] = 1.0 / (centrality + 0.01)

karate_weighted.precomputing()
print(f"Weighted karate club graph created successfully")

    Weighted karate club graph created successfully

    Example 4: Parameter Sensitivity Analysis¶

    Understanding how parameters affect clustering:

    from kmeanssa_ng import generate_simple_graph

# Create a symmetric test graph
simple_graph = generate_simple_graph(n_a=10, n_aa=5, bridge_length=3.0)
simple_graph.precomputing()
simple_points = simple_graph.sample_points(50)

print(f"Simple test graph: {simple_graph.number_of_nodes()} nodes")

    Simple test graph: 122 nodes

    # Test different parameter combinations
parameters = [
    {"lambda_param": 0.5, "beta": 2.0, "name": "Low exploration, high drift"},
    {"lambda_param": 1.0, "beta": 1.0, "name": "Balanced"},
    {"lambda_param": 2.0, "beta": 0.5, "name": "High exploration, low drift"},
]

results = []
for params in parameters:
    qg_sa_test = QGSimulatedAnnealing(
        simple_points, k=2, 
        lambda_param=params["lambda_param"], 
        beta=params["beta"]
    )
    node_centers_test = qg_sa_test.run_for_kmeans(robust_prop=0.1)

    # Calculate energy using base class for comparison
    sa_test = SimulatedAnnealing(simple_points, k=2, 
                                lambda_param=params["lambda_param"], 
                                beta=params["beta"])
    centers_test = sa_test.run(initialization="kpp")
    energy = sa_test.calculate_energy(centers_test, simple_points)

    results.append({
        "name": params["name"],
        "energy": energy,
        "node_centers": node_centers_test
    })

# Compare results
print("Parameter sensitivity analysis:")
for result in results:
    print(f"{result['name']}: Energy = {result['energy']:.3f}, Nodes = {result['node_centers']}")

    Parameter sensitivity analysis:
Low exploration, high drift: Energy = 7.085, Nodes = ['B10', 'A6']
Balanced: Energy = 4.218, Nodes = ['A0', 'B84']
High exploration, low drift: Energy = 5.005, Nodes = ['B0', 'A0']

    Example 6: Batch Processing and Reproducibility¶

    Processing multiple graphs and ensuring reproducible results:

    import numpy as np
from kmeanssa_ng import generate_random_sbm

# Set random seed for reproducibility
np.random.seed(42)

# Process multiple random graphs
results = []
for trial in range(3):  # Reduced for demo
    # Generate random SBM with different edge lengths
    graph_random = generate_random_sbm(
        sizes=[25, 25], 
        p=[[0.8, 0.1], [0.1, 0.8]],
        lengths=[[1.0, 3.0], [3.0, 1.0]]  # Different intra/inter lengths
    )

    graph_random.precomputing()
    points_random = graph_random.sample_points(80)

    # Use quantum graph method
    qg_sa_random = QGSimulatedAnnealing(points_random, k=2)
    node_centers_random = qg_sa_random.run_for_kmeans(robust_prop=0.1)

    # Calculate energy for comparison
    sa_random = SimulatedAnnealing(points_random, k=2)
    centers_random = sa_random.run(initialization="kpp")
    energy = sa_random.calculate_energy(centers_random, points_random)

    results.append({
        "trial": trial + 1,
        "energy": energy,
        "node_centers": node_centers_random
    })

# Analyze results
energies = [r["energy"] for r in results]
print(f"Energy statistics across {len(results)} trials:")
print(f"  Mean: {np.mean(energies):.3f}")
print(f"  Std:  {np.std(energies):.3f}")
print(f"  Min:  {np.min(energies):.3f}")
print(f"  Max:  {np.max(energies):.3f}")

    Energy statistics across 3 trials:
  Mean: 1.463
  Std:  0.044
  Min:  1.400
  Max:  1.500

    Quick Setup for Large Graphs¶

    # For large graphs, optimize for speed
# Using the existing graph for demonstration
print("Performance optimization example:")

# Use larger steps and less robustification
sa_fast = QGSimulatedAnnealing(points, k=2, step_size=0.2)
centers_fast = sa_fast.run_for_kmeans(robust_prop=0.05)
print(f"Fast clustering centers: {centers_fast}")

    Performance optimization example:
Fast clustering centers: [48, 52]

    Memory Considerations¶

    # For very large graphs, be careful with point sampling
n_points = min(100, len(graph.edges()) * 2)  # Conservative limit for demo
print(f"Recommended points for this graph: {n_points}")
print(f"Graph has {len(graph.edges())} edges")

    Recommended points for this graph: 100
Graph has 1377 edges

    Debugging Clustering Issues¶

    # If clustering seems poor, try these diagnostics:

# 1. Check graph connectivity
print(f"Graph connected: {nx.is_connected(graph)}")
print(f"Number of components: {nx.number_connected_components(graph)}")

# 2. Try different parameters
test_params = [
    {"lambda_param": 0.5, "name": "Conservative"},
    {"lambda_param": 2.0, "name": "Exploratory"}
]

print("\nParameter comparison:")
for params in test_params:
    sa_debug = QGSimulatedAnnealing(points[:50], k=2, lambda_param=params["lambda_param"])
    centers_debug = sa_debug.run_for_kmeans(robust_prop=0.1)
    print(f"  {params['name']} (λ={params['lambda_param']}): {centers_debug}")

    Graph connected: True
Number of components: 1

Parameter comparison:
  Conservative (λ=0.5): [59, 37]
  Exploratory (λ=2.0): [61, 28]
